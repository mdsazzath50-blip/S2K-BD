<script>
const app = {
    payNumbers: { bkash: "01606287039", nagad: "01826191236" },
    
    db: JSON.parse(localStorage.getItem('C2K_VIP_PRO_DB')) || {
        isLoggedIn: false,
        name: "", phone: "", balance: 0,
        vip: 0, vipExpiry: null,
        id: Math.floor(Math.random() * 90000) + 10000,
        tasksDone: [],
        lastSpin: 0,
        currentTaskPart: 1
    },

    init: function() {
        this.checkVipExpiry();
        if(this.db.isLoggedIn) {
            document.getElementById('login-overlay').style.display = 'none';
            document.getElementById('main-app').style.display = 'flex';
            this.updateUI();
            this.renderVIPs();
            this.renderTasks();
        }
    },

    login: function() {
        const name = document.getElementById('log-name').value;
        const phone = document.getElementById('log-phone').value;
        if(name.length < 3 || phone.length < 11) return Swal.fire('Error', 'Invalid info!', 'error');
        this.db.name = name; this.db.phone = phone; this.db.isLoggedIn = true;
        this.save(); this.init();
    },

    save: function() {
        localStorage.setItem('C2K_VIP_PRO_DB', JSON.stringify(this.db));
        this.updateUI();
    },

    checkVipExpiry: function() {
        if (this.db.vip > 0 && this.db.vipExpiry) {
            if (Date.now() > this.db.vipExpiry) {
                this.db.vip = 0;
                this.db.vipExpiry = null;
                this.save();
                Swal.fire('Expired', 'Your VIP validity has ended!', 'warning');
            }
        }
    },

    updateUI: function() {
        document.getElementById('user-balance').innerText = Math.floor(this.db.balance);
        document.getElementById('prof-name').innerText = this.db.name;
        document.getElementById('prof-phone').innerText = this.db.phone;
        document.getElementById('ref-code-display').innerText = this.db.id;
        
        const vBadge = document.getElementById('header-vip');
        vBadge.innerText = this.db.vip > 0 ? "VIP " + this.db.vip : "BASIC";
        vBadge.className = "vip-badge v" + this.db.vip;

        const expiryBox = document.getElementById('expiry-box');
        if(this.db.vip > 0 && this.db.vipExpiry) {
            const daysLeft = Math.ceil((this.db.vipExpiry - Date.now()) / (1000 * 60 * 60 * 24));
            expiryBox.innerHTML = `<span class="expiry-text">Validity: ${daysLeft} Days Left</span>`;
        } else {
            expiryBox.innerHTML = "";
        }

        // VIP হলে ২০০ কাজ, না হলে ৩০ কাজ
        const limit = this.db.vip > 0 ? 200 : 30;
        document.getElementById('task-count').innerText = `${this.db.tasksDone.length}/${limit} Done`;
    },

    nav: function(scr, el) {
        document.querySelectorAll('.screen').forEach(s => s.classList.remove('active-screen'));
        document.getElementById('scr-' + scr).classList.add('active-screen');
        document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
        if(el.classList.contains('nav-item')) el.classList.add('active');
    },

    getReward: function() {
        const rewards = { 0:2, 1:10, 2:25, 3:50, 4:100, 5:150, 6:250, 7:400, 8:1000 };
        return rewards[this.db.vip] || 2;
    },

    renderTasks: function() {
        const container = document.getElementById('task-container');
        const nav = document.getElementById('task-sections-nav');
        container.innerHTML = ""; nav.innerHTML = "";
        
        // লিমিট সেট করা হচ্ছে
        let totalLimit = this.db.vip > 0 ? 200 : 30;
        let parts = this.db.vip > 0 ? 4 : 1; 
        let perPart = totalLimit / parts;
        
        if(this.db.vip > 0) {
            for(let p=1; p<=parts; p++) nav.innerHTML += `<button class="btn btn-sm ${this.db.currentTaskPart == p ? '' : 'v2'}" onclick="app.setPart(${p})">Part ${p}</button>`;
        }
        let start = (this.db.currentTaskPart - 1) * perPart + 1;
        let end = this.db.currentTaskPart * perPart;
        for(let i=start; i<=end; i++) {
            const isDone = this.db.tasksDone.includes(i);
            container.innerHTML += `<div class="task-card ${isDone ? 'completed' : ''}"><p>Task #${i}</p><b>৳ ${this.getReward()}</b><button class="btn btn-sm" onclick="app.doTask(${i})">${isDone ? 'Done' : 'Start'}</button></div>`;
        }
    },

    setPart: function(p) { this.db.currentTaskPart = p; this.renderTasks(); },

    doTask: function(id) {
        // টাস্ক লিমিট চেক
        let totalLimit = this.db.vip > 0 ? 200 : 30;
        if(this.db.tasksDone.length >= totalLimit) return Swal.fire('Limit Over', 'আজকের কাজের লিমিট শেষ!', 'warning');

        Swal.fire({ title: 'Loading Ad...', timer: 3000, showConfirmButton: false, didOpen: () => Swal.showLoading() }).then(() => {
            this.db.balance += this.getReward();
            this.db.tasksDone.push(id);
            this.save(); this.renderTasks();
            Swal.fire('Success', 'Money Added!', 'success');
        });
    },

    renderVIPs: function() {
        const vips = [
            { lv: 1, price: 500, info: "৳10/Task" },
            { lv: 2, price: 1000, info: "৳25/Task" },
            { lv: 3, price: 2000, info: "৳50/Task" },
            { lv: 8, price: 15000, info: "৳1000/Task" }
        ];
        document.getElementById('vip-list').innerHTML = vips.map(v => `
            <div class="card">
                <div style="display:flex; justify-content:space-between;"><b>VIP Level ${v.lv}</b><span class="vip-badge v${v.lv}">৳ ${v.price}</span></div>
                <p style="font-size:0.7rem;">${v.info} | Valid for 30 Days</p>
                <button class="btn" onclick="app.buyVIPModal(${v.lv}, ${v.price})">Upgrade</button>
            </div>`).join('');
    },

    buyVIPModal: function(lv, price) {
        Swal.fire({
            title: `Upgrade VIP ${lv}`,
            text: `Send ৳${price} to ${this.payNumbers.bkash} and enter TRX ID.`,
            input: 'text', inputPlaceholder: 'Transaction ID'
        }).then((res) => {
            if(res.value) {
                const expiry = new Date();
                expiry.setDate(expiry.getDate() + 30); // 30 Days Validity
                this.db.vip = lv;
                this.db.vipExpiry = expiry.getTime();
                this.db.tasksDone = [];
                this.save(); this.renderTasks();
                Swal.fire('Success', 'VIP Active for 30 Days!', 'success');
            }
        });
    },

    spinWheel: function() {
        const now = Date.now();
        if(now - this.db.lastSpin < 60000) return Swal.fire('Wait', '1 min delay', 'warning');
        document.getElementById('wheel').style.transform = `rotate(${Math.floor(Math.random() * 360) + 720}deg)`;
        setTimeout(() => {
            const win = Math.floor(Math.random() * 10) + 2;
            this.db.balance += win; this.db.lastSpin = Date.now(); this.save();
            Swal.fire('Won', `৳ ${win}`, 'success');
        }, 3000);
    },

    claimDaily: function() {
        const today = new Date().toDateString();
        if(this.db.lastDaily === today) return Swal.fire('Error', 'Already done!', 'error');
        this.db.balance += (this.db.vip > 0 ? 50 : 10);
        this.db.lastDaily = today; this.save(); Swal.fire('Success', 'Bonus Added!', 'success');
    },

    requestWithdraw: function() {
        const amt = parseInt(document.getElementById('wd-amount').value);
        if(!amt || amt < 500) return Swal.fire('Error', 'Min 500!', 'error');
        if(amt > this.db.balance) return Swal.fire('Error', 'No balance!', 'error');
        this.db.balance -= amt; this.save(); Swal.fire('Success', 'Pending!', 'success');
    },

    logout: function() { localStorage.clear(); location.reload(); }
};
app.init();
</script>
